def read(io)
  lock io do
    @key  = read_bytes(io)
    @data = read_bytes(io)
    crc   = io.read(4)
    raise CorruptDataError, "CRC mismatch #{crc} should be #{crc_string}" unless crc == crc_string
  end
  self
end
